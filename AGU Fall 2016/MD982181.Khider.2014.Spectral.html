A Jupyter Notebook for Spectral Analysis of time-uncertain Marine Data¶

Introduction ¶

This notebook works through the step of obtaining an ensemble of possible age models for the marine chronology and obtain the time-uncertain spectrum for the time series.

**NOTE**

Please cite the follwing contributors for Bchron and the z-wavelet

Haslett, J., Parnell, A. (2008). A simple monotone process with application to radiocarbon-dated depth chronologies. Journal of the Royal Statistical Society: Series C (Applied Statistics), 57, 4, 1467-9876. doi: 10.1111/j.1467-9876.2008.00623.x

Running this Notebook ¶

Software¶

  1. Jupyter Notebook
  2. Python 3.4+ with the following modules:
    1. lipd 1.2+ (pip install lipd)
    2. numpy (standard with Anaconda Install)
    3. pandas (standard with Anaconda Install)
    4. matplotlib (standard with Anaconda Install)
  3. An up-to-date version of R installed on your computer
  4. The Bchron package. In a R console, type install.packages('Bchron')

def findLipds(path):
    """
    Find all the lipd files in a directory
    """

    files = os.listdir(path)
    lipd_in_directory = [i for i in files if i.endswith('.lpd')]

    return lipd_in_directory

def createdir(path, foldername):
    """
    create a new folder in the working directory
    """

    if not os.path.exists(path+'/'+foldername):
        os.makedirs(path+'/'+foldername)
    
    newdir = path+'/'+foldername    

    return newdir  

def enumerateLipds(path):
    """
    enumerate the LiPDs in the directory and allows to select one
    """
    lipd_in_directory = findLipds(path)
    print("Below are the available records")
    for idx, val in enumerate(lipd_in_directory):
        print(idx,': ',val)   

def promptforLipd():
    """
    Ask the user to select a LiPD file from a list
    Use this function in conjunction with enumerateLipds()
    """
    select_lipd = int(input("Enter the number of the file you wish to analyze: "))
    return select_lipd 

def readHeaders(Dataframe):
    """
    read and returns the column headers of a panda Dataframe
    """
    index = 0
    for key, value in Dataframe.items():
        print(index,": ",key)
        index+=1

def promptforVariable():
    """
    Ask the user to select the variable they are interested in.
    Use this function in conjunction with readHeaders()
    """
    select_var = int(input("Enter the number of the variable you wish to use: ")) 
    return select_var

def selectVariable(Dataframe, varName):
    """
    Select a specific column in a panda dataframes based on the variable name
    in the LinkedEarth Ontology
    Inputs: the dataframe and the name of the variable to search for
    """
    index =0 #index header
    index_option=[]# keep the index
    option = [] #number of columns meeting the varName criteria
    for key,val in Dataframe.items():
        if varName in key.lower():
            index_option.append(index)
            option.append(key)
        index+=1    
    # if there is only one option, select it by default
    index = 0 #reset the counter
    if len(option)==1:
        var_idx= index_option[0]
        label = option[0]
    elif len(option)==0:
        print("Only the following variables are available: ")
        readHeaders(Dataframe)
        var_idx = promptforVariable()
        label = Dataframe.columns.values[var_idx]        
    else:
        print("There is more than one option available")
        for val in option:        
            print(index,": ",val)
            index+=1
        option_index = int(input("Enter the index of the variable of interest: "))
        var_idx = index_option[option_index]
        label = option[option_index]

    return var_idx, label
    

def promptforXaxis(Dataframe):
    """
    read the column headers and try to determine the choice for x-axis
    """
    index =0 #index header
    index_option=[]# keep the index
    x_option = [] #number of depth/age column
    for key,val in Dataframe.items():
        if 'age' in key.lower() or 'depth' in key.lower():
            index_option.append(index)
            x_option.append(key)
        index+=1    
    # if there is only one option, select it by default
    index = 0 #reset the counter
    if len(x_option)==1:
        x_axis_idx= index_option[0]
        x_axis_label = x_option[0]
    else:
        print("There is more than one option available")
        for val in x_option:        
            print(index,": ",val)
            index+=1
        x_index = int(input("Enter the index for the x-axis representation: "))
        x_axis_idx = index_option[x_index]
        x_axis_label = x_option[x_index]
    
    return x_axis_idx, x_axis_label
                      
def getDataframe(filename="", refine="paleo", tablenumber=[]): 
    """
    Get the data to be analyzed in the proper dataframe
    Inputs:
        - filename: if unknown, will prompt. Need the .lpd extension
        - refine: which data are you interested in. Two options: paleo, chron. Default is paleo.
        - tablenumber: if there are multiple datatables, enter the number of the table. Default is unknown
    """ 
    # Get the name of the LiPD file
    if not filename:
        lipd_in_directory=findLipds(lpd.path)
        enumerateLipds(lpd.path)
        filename = lipd_in_directory[promptforLipd()]
    
    # Get the names of the                                  
    c = lpd.getCsv(filename)
    
    tablename=[]
    for key, value in c.items():
        if refine in key.lower():
            tablename.append(key)
    if len(tablename) == 1:
        tablename = tablename[0]
    elif len(tablename) == 0:
        sys.exit("No table match your selection criteria")
    else:
        if not tablenumber:
            print("More than one table match the search criteria")
            for idx, val in enumerate(tablename):
                print(idx,": ",val)
            tablenumber=int(input("Which one would you like to use? "))
        tablename=tablename[tablenumber]
                    
    dfs_lipd = lpd.lipdToDf(filename)
    DataFrame = dfs_lipd[refine+"Data"][tablename]                 
    
    return DataFrame, filename 

def stopFunction(answer,keyword, message):

    and return an error message.
    """
    if answer == keyword:
        sys.exit(message)
        
def recordID(filename=""):
    """
    Get basic information about a record:
        - name
        - archiveType
        - Authors
    """  
    if not filename:
        lipd_in_directory=findLipds(lpd.path)
        enumerateLipds(lpd.path)
        filename = lipd_in_directory[promptforLipd()]
    
    # get the medatadata  
    lipd_m = lpd.getMetadata(filename)
    name = lipd_m['dataSetName'] # get the name 
    archivetype = lipd_m['archiveType'] # get the archive type
    authors_list=[]
    # Loop through the list of contributors
    for key,val in enumerate(lipd_m['investigator']):
        authors_list.append(val['name'])
    # Convert to a string
    authors = ', '.join(str(e) for e in authors_list)
    if len(authors_list)>1:
        authors_short = authors_list[0]+ " et al."
    else:   
        authors_short = authors                      
    return name, archivetype, authors, authors_short 

def get_recursively(search_dict, field):
    """Takes a dict with nested lists and dicts,
    and searches all dicts for a key of the field
    provided.
    """
    fields_found = []

    for key, value in search_dict.items():

        if key == field:
            fields_found.append(value)

        elif isinstance(value, dict):
            results = get_recursively(value, field)
            for result in results:
                fields_found.append(result)

        elif isinstance(value, list):
            for item in value:
                if isinstance(item, dict):
                    more_results = get_recursively(item, field)
                    for another_result in more_results:
                        fields_found.append(another_result)

    return fields_found
    
def valuesloc(filename="", refine="paleo", tablenumber=[], varName=""):
    """
    Look for the indexes where there are no missing values for the variable 
    """
    # Get the dataframe
    dataframe, filename = getDataframe(filename=filename, refine = refine, tablenumber=tablenumber)
    # Now get the metadata dictionary
    lipd_m = lpd.getMetadata(filename)
    # Get the missing values handle
    value = get_recursively(lipd_m, "missingValue")
    
    # Check that there is only one possible handle for missing values, if not
    # prompt the users for the right one.
    unique=[] # create the unique list
    for i in value:
        if i not in unique:
            unique.append(i)
    if len(unique)>1:
        print("Missing values were flagged in the following ways: ")
        for key, val in enumerate(unique):
            print(key,": ", val)
        answer = int(input("Which one would you like to use? "))
        missing_value = unique[answer]
    else:
        missing_value = unique[0]
    
    # if the variable name is not given or doesn't exist prompt
    if not varName:
        print("Below is a list of variables: ")
        readHeaders(dataframe)
        var_idx = promptforVariable()
    else:
        var_idx, label = selectVariable(dataframe, varName)
    
    # Get the index with the appropriate values
    val_idx = [] #Initiate the counter
    if missing_value == 'NaN':
        for n,i in enumerate(dataframe.iloc[:,var_idx]):
            if not np.isnan(i):
                val_idx.append(n)
    else:
        for n,i in enumerate(dataframe.iloc[:,var_idx]):
            if i!=float(missing_value):
                val_idx.append(n)

    return val_idx

# Lomb-Scargle Periodogram

ofac = 4 # oversampling value
hifac = 1 # highest frequency that can be analyzed/Nyquist frequency
int = np.nanmean(np.diff(mquantiles(age_paleo_nan, prob=[0.5], axis=0)))
step = ((2*int)**(-1))/(age_paleo_nan.shape[1]*ofac) # step size for the frequency vector
f = np.arange(2*np.pi*step, 2*np.pi*(hifac*(2*int)**(-1)), step)

# compute the periodogram for each age model realization

pgram = np.zeros((age_paleo_nan.shape[0],f.shape[0])) # pre-allocate the space   
                 
for index in np.arange(0,age_paleo_nan.shape[0],1):
    pgram[index,:] = signal.lombscargle(age_paleo_nan[index,:],y_values_std,f)

pgram_norm = np.sqrt(4*(pgram/y_values.shape[0]))
pgramQ  = mquantiles(pgram_norm, prob=[0.025,0.975], axis=0)

# Find the peaks for each Monte-Carlo age models and the percent variance in each band of interest
from detect_peaks import detect_peaks
from scipy.integrate import simps

# initiate the matrices
peak_sub1 = np.zeros(age_paleo_nan.shape[0])
peak_sub2 = np.zeros(age_paleo_nan.shape[0])
peak_sub3 = np.zeros(age_paleo_nan.shape[0])

percent_var1 = np.zeros(age_paleo_nan.shape[0])
percent_var2 = np.zeros(age_paleo_nan.shape[0])
percent_var3 = np.zeros(age_paleo_nan.shape[0])

for index in np.arange(0,age_paleo_nan.shape[0],1): #step over all the possible spectres 
    # Concentrate on the 900-1200 yr band
    pgram_sub1 = pgram_norm[index , np.min(np.where(2*np.pi/f<1200)):np.max(np.where(2*np.pi/f>900))]
    f_sub1 = f[np.min(np.where(2*np.pi/f<1200)):np.max(np.where(2*np.pi/f>900))]
    ind_sub1 = detect_peaks(pgram_sub1, show = False, mpd = pgram_sub1.shape[0])
    if not ind_sub1:
        peak_sub1[index] = np.nan
        percent_var1[index]= np.nan
    else:    
        peak_sub1[index] = 2*np.pi/f_sub1[ind_sub1]
        percent_var1[index] = simps(pgram_sub1,f_sub1)/simps(pgram_norm[index,:], f)

    # Concentrate on the 1200-2000 yr band
    pgram_sub2 = pgram_norm[index , np.min(np.where(2*np.pi/f<2000)):np.max(np.where(2*np.pi/f>1200))]
    f_sub2 = f[np.min(np.where(2*np.pi/f<2000)):np.max(np.where(2*np.pi/f>1200))]
    ind_sub2 = detect_peaks(pgram_sub2, show = False, mpd = pgram_sub2.shape[0])
    if not ind_sub2:
        peak_sub2[index] = np.nan
        percent_var2[index]= np.nan
    else:    
        peak_sub2[index] = 2*np.pi/f_sub2[ind_sub2]
        percent_var2[index] = simps(pgram_sub2,f_sub2)/simps(pgram_norm[index,:], f)

    # Concentrate on the 2000-3000 yr band
    pgram_sub3 = pgram_norm[index , np.min(np.where(2*np.pi/f<3000)):np.max(np.where(2*np.pi/f>2000))]
    f_sub3 = f[np.min(np.where(2*np.pi/f<3000)):np.max(np.where(2*np.pi/f>2000))]
    ind_sub3 = detect_peaks(pgram_sub3, show = False, mpd = pgram_sub3.shape[0])
    if not ind_sub3:
        peak_sub3[index] = np.nan
        percent_var3[index]= np.nan
    else:    
        peak_sub3[index] = 2*np.pi/f_sub3[ind_sub3]
        percent_var3[index] = simps(pgram_sub3,f_sub3)/simps(pgram_norm[index,:], f)
